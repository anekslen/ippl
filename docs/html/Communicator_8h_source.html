158  template <class Buffer, typename Archive>
159  void recv(int src, int tag, Buffer& buffer, Archive& ar, size_type msize,
160  size_type nrecvs) {
161  // Temporary fix. MPI communication seems to have problems when the
162  // count argument exceeds the range of int, so large messages should
163  // be split into smaller messages
164  if (msize > INT_MAX) {
165  std::cerr << "Message size exceeds range of int" << std::endl;
166  this->abort();
167  }
168  MPI_Status status;
169  MPI_Recv(ar.getBuffer(), msize, MPI_BYTE, src, tag, *comm_m, &status);
170 
171  buffer.deserialize(ar, nrecvs);
172  }

176  size_type nsends) {
