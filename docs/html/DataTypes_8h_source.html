66  MPI_Datatype type = MPI_BYTE;
67  if (core::type_names.find(std::type_index(typeid(T))) == core::type_names.end()) {
68  if constexpr(vector_dim_type<T>::Dim > 0) {
69  MPI_Datatype tp;
70  MPI_Type_contiguous(vector_dim_type<T>::Dim, get_mpi_datatype<typename T::value_type>(typename T::value_type{}), &tp);
71  MPI_Type_commit(&tp);
72  core::type_names[std::type_index(typeid(T))] = tp;
73  }
74  }
75  try {
